Understanding Callbacks

A callback is a function passed as an argument to another function.

// Simple callback
function greet(name, callback) {
    console.log("Hello, " + name);
    callback();
}

greet("Alice", () => {
    console.log("Callback executed!");
});

// Callback with data
function fetchUser(id, callback) {
    setTimeout(() => {
        const user = { id, name: "User " + id };
        callback(user);
    }, 1000);
}

fetchUser(1, (user) => {
    console.log(user);  // { id: 1, name: "User 1" }
});
⚠️ Callback Hell: Too many nested callbacks make code hard to read. Promises and async/await solve this.

Promises

A Promise represents a value that may not be available yet but will be resolved in the future.

Creating Promises

// Basic promise
const promise = new Promise((resolve, reject) => {
    setTimeout(() => {
        resolve("Success!");
    }, 1000);
});

// Consuming the promise
promise.then((result) => {
    console.log(result);  // "Success!" (after 1 second)
});

// With rejection
const fetchData = new Promise((resolve, reject) => {
    const success = false;
    
    if (success) {
        resolve("Data loaded");
    } else {
        reject("Failed to load data");
    }
});

fetchData
    .then(result => console.log(result))
    .catch(error => console.error(error));  // Handles rejection

Promise Chaining

fetch("https://api.example.com/user/1")
    .then(response => response.json())
    .then(user => {
        console.log(user);
        return fetch("https://api.example.com/posts/" + user.id);
    })
    .then(response => response.json())
    .then(posts => console.log(posts))
    .catch(error => console.error("Error:", error))
    .finally(() => console.log("Done!"));

Promise.all()

const promise1 = fetch("/api/users");
const promise2 = fetch("/api/posts");
const promise3 = fetch("/api/comments");

Promise.all([promise1, promise2, promise3])
    .then(([users, posts, comments]) => {
        console.log("All data loaded!");
    })
    .catch(error => console.error("One or more failed"));

Async/Await

Modern syntax for handling asynchronous code. Cleaner than promises!

Basic Async/Await

// Async function returns a Promise
async function getData() {
    const response = await fetch("https://api.example.com/data");
    const data = await response.json();
    return data;
}

// Call async function
getData().then(data => console.log(data));

Error Handling with Try/Catch

async function loadUser(id) {
    try {
        const response = await fetch(`/api/users/${id}`);
        
        if (!response.ok) {
            throw new Error("User not found");
        }
        
        const user = await response.json();
        console.log(user);
    } catch (error) {
        console.error("Error:", error.message);
    } finally {
        console.log("Request completed");
    }
}

loadUser(1);

Parallel Requests

async function loadMultiple() {
    try {
        // Sequential (slow)
        const user = await fetch("/api/user").then(r => r.json());
        const posts = await fetch("/api/posts").then(r => r.json());
        
        // Parallel (fast)
        const [userData, postsData] = await Promise.all([
            fetch("/api/user").then(r => r.json()),
            fetch("/api/posts").then(r => r.json())
        ]);
        
        console.log("Both loaded simultaneously!");
    } catch (error) {
        console.error(error);
    }
}

loadMultiple();

Async/Await in Loops

async function processItems(ids) {
    // Sequential processing
    for (const id of ids) {
        const item = await fetch(`/api/item/${id}`).then(r => r.json());
        console.log(item);
    }
    
    // Parallel processing
    const items = await Promise.all(
        ids.map(id => fetch(`/api/item/${id}`).then(r => r.json()))
    );
    console.log(items);
}

processItems([1, 2, 3, 4, 5]);
